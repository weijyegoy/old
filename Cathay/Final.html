# 特徵縮放
from sklearn.preprocessing import StandardScaler

# 選擇需要縮放的數值型特徵
numeric_features = ['person_age', 'person_income', 'person_emp_exp', 
                   'loan_amnt', 'loan_int_rate', 'loan_percent_income',
                   'cb_person_cred_hist_length', 'credit_score']

# 創建StandardScaler對象
scaler = StandardScaler()

# 對數值型特徵進行縮放
X[numeric_features] = scaler.fit_transform(X[numeric_features])

# 使用SelectKBest進行特徵選擇
selector = SelectKBest(score_func=f_classif, k=10)

# 在X_selected = selector.fit_transform(X, y)之前加入：

# 1. 創建交互特徵
X['income_to_loan'] = X['person_income'] / X['loan_amnt']
X['credit_to_income'] = X['credit_score'] / X['person_income']
X['age_to_experience'] = X['person_age'] / (X['person_emp_exp'] + 1)

# 2. 特徵標準化
scaler = StandardScaler()
X[numeric_features] = scaler.fit_transform(X[numeric_features])


X_selected = selector.fit_transform(X, y)

# 獲取選擇的特徵索引
selected_features_mask = selector.get_support()
selected_features = X.columns[selected_features_mask].tolist()

# 打印選擇的特徵
print("選擇的特徵數量：", len(selected_features))
print("\n選擇的特徵：")
for feature, score in zip(X.columns[selected_features_mask], selector.scores_[selected_features_mask]):
    print(f"{feature}: {score:.2f}")

# 使用選擇的特徵創建新的特徵矩陣
X_selected = X[selected_features]

# 再次分割數據
X_train, X_test, y_train, y_test = train_test_split(X_selected, y, test_size=0.2, random_state=42)

    'Random Forest': RandomForestClassifier(n_estimators=100, random_state=42),
    'Logistic Regression': LogisticRegression(random_state=42),
    'Decision Tree': DecisionTreeClassifier(random_state=42),
    'XGBoost': xgb.XGBClassifier(random_state=42),
    'Naive Bayes': GaussianNB()
}

# 存儲各個模型的評估結果
results = {}

for name, model in models.items():
    # 訓練模型
    model.fit(X_train_balanced, y_train_balanced)
    
    # 預測
    y_pred = model.predict(X_test)
    y_pred_proba = model.predict_proba(X_test)[:, 1]

    results[name] = {
        'Accuracy': accuracy_score(y_test, y_pred),
        'Precision': precision_score(y_test, y_pred),
        'Recall': recall_score(y_test, y_pred),
        'F1': f1_score(y_test, y_pred),
        'AUC-ROC': roc_auc_score(y_test, y_pred_proba)
